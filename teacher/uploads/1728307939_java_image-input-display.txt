import javax.swing.*;
import java.awt.*;
import java.io.File;
import java.util.Scanner;

public class PictureScanner {
    public static void main(String[] args) {
        // Create a Scanner object to read input
        Scanner scanner = new Scanner(System.in);

        // Ask the user for the file path of the picture
        System.out.print("Enter the file path of the picture: ");
        String filePath = scanner.nextLine();

        // Simulate scanning the picture
        System.out.println("Scanning picture at: " + filePath);

        // Display the picture
        displayImage(filePath);

        // Close the scanner
        scanner.close();
    }

    private static void displayImage(String filePath) {
        // Create a JFrame to display the image
        JFrame frame = new JFrame("Picture Scanner");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setSize(600, 600); // Adjust the size as needed

        // Check if the file exists
        File file = new File(filePath);
        if (!file.exists() || !file.isFile()) {
            System.out.println("File not found: " + filePath);
            return;
        }

        // Load the image
        ImageIcon imageIcon = new ImageIcon(filePath);

        // Resize the image to fit the frame if necessary
        Image image = imageIcon.getImage();
        Image resizedImage = image.getScaledInstance(frame.getWidth(), frame.getHeight(), Image.SCALE_SMOOTH);
        imageIcon = new ImageIcon(resizedImage);

        // Create a JLabel to hold the image
        JLabel label = new JLabel(imageIcon);
        frame.add(label);

        // Display the JFrame
        frame.setVisible(true);
    }
}
